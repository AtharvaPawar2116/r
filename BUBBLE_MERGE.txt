#include <iostream>
#include <omp.h>
using namespace std;

// Parallel Bubble Sort using Odd-Even Transposition
void parallelBubbleSort(int arr[], int n) {

    for (int phase = 0; phase < n; phase++) {

        // Even Phase
        if (phase % 2 == 0) {

            #pragma omp parallel for
            for (int i = 1; i < n; i += 2) {

                if (arr[i - 1] > arr[i]) {
                    swap(arr[i - 1], arr[i]);
                }
            }
        }

        // Odd Phase
        else {

            #pragma omp parallel for
            for (int i = 1; i < n - 1; i += 2) {

                if (arr[i] > arr[i + 1]) {
                    swap(arr[i], arr[i + 1]);
                }
            }
        }
    }
}

int main() {

    int n;

    cout << "Enter number of elements: ";
    cin >> n;

    int arr[n];

    cout << "Enter elements:\n";

    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    parallelBubbleSort(arr, n);

    cout << "Sorted Array:\n";

    for (int i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }

    return 0;
}
